#include <stdio.h>
#include <math.h>
int main() {
   int n, isPrime = 1;
   printf("Enter a number: ");
   scanf("%d", &n);
   if (n <= 1) {
       isPrime = 0;
   } else {
       for (int i = 2; i <= sqrt(n); i++) {
           if (n % i == 0) {
               isPrime = 0;
               break;
           }
       }
   }
   if (isPrime)
       printf("%d is a Prime Number\n", n);
   else
       printf("%d is NOT a Prime Number\n", n);
   return 0;
}
